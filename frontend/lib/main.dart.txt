import 'package:flutter/material.dart';
import 'screens/home_screen.dart';
import 'screens/voice_selection.dart';
import 'screens/ambiance_selection.dart';
import 'screens/instrument_selection.dart';

void main() {
  runApp(EurydiceApp());
}

class EurydiceApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      title: 'Eurydice - Générateur de chansons',
      theme: ThemeData(
        primarySwatch: Colors.deepPurple,
        scaffoldBackgroundColor: Colors.white,
      ),
      initialRoute: '/',
      routes: {
        '/': (context) => HomeScreen(),
        '/voices': (context) => VoiceSelection(),
        '/ambiances': (context) => AmbianceSelection(),
        '/instruments': (context) => InstrumentSelection(),
      },
    );
  }
}
